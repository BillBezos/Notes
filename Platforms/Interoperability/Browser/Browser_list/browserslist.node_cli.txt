                                  #  - version 'vA.B.C'
                                  #  - date 'YYYY-MM-DD'
                                  #  - npm|uv|zlib 'A.B.C'|undefined

                                  #  - start 'YYYY-MM-DD': release date, i.e. -> current
                                  #     - might be future, i.e. not released yet
                                  #  - lts 'YYYY-MM-DD': LTS-current -> LTS-active
                                  #  - maintainance 'YYYY-MM-DD': LTS-active|non-LTS-current -> maintained
                                  #  - end 'YYYY-MM-DD': maintained -> not maintained
                                  #  - codename STR: LTS name
                                  #     - can be '' (future LTS)
                                  #     - undefined means not a LTS
                                  #MAJOR_VERSION is 'v0.10|12' or 'v4|5|6|...'

                                  #  - defaults (def), i.e. '> 0.5%, last 2 versions, Firefox ESR, not dead'
                                  #  - last NUM [major] [BROWSER_NAME] versions
                                  #  - BROWSER_NAME ESR: same as last 1 BROWSER_NAME versions
                                  #  - unreleased [BROWSER_NAME] versions
                                  #  - BROWSER_NAME [> >= < <=] VERSION[-VERSION2]
                                  #  - maintained node versions
                                  #  - current node
                                  #  - >|>=|<|<= NUM% [in COUNTRY|my stats]
                                  #  - cover NUM% [in COUNTRY|my stats]
                                  #  - since YYYY[-MM][-DD]
                                  #  - last NUM years
                                  #  - extends PATH|PACKAGE (exporting a STR_ARR)
                                  #  - dead (last 2 versions but no usage and no company support, e.g. IE10 or Blackberry 7)
                                  #  - not BROWSER

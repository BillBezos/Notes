                                  #  - templatePath 'TEMPLATE_DIR' (def: library default templates):
                                  #     - directory containing tests templates
                                  #     - can be changed to customize templates
                                  #  - testModule 'request|superagent': HTTP request module
                                  #  - assertionFormat 'should|expect|assert': assertion module
                                  #  - pathParams, requestData, maxLen: see below
                                  #  - loadTest OBJ_ARR (def: none):
                                  #     - generates load tests with module "arete"

Template TEMPLATE_DIR/TEST_MODULE/METHOD/METHOD.handlebars
Template data:
  - responseCode 'STATUS_CODE'
  - default BOOL (whether 'STATUS_CODE' is 'default')
  - description 'STATUS_CODE RESP.description'
  - assertion OPTS.assertionFormat
  - requestData OPTS.requestData./PATH.METHOD.STATUS_CODE REQ_DATA_ARR:
     - REQ_DATA can be:
        - { name: 'PATH_PARAM_NAME', description STR } (path PARAM)
        - { body: OBJ, description STR } (body PARAM)
        - { 'PARAM_NAME': VAL, description STR } (other PARAMs)
     - METHOD.handlebars template will be run once per REQ_DATA
     - sets TEMPLATE_DATA:
        - pathParams REQ_DATA (unless OPTS.pathParams defined)
        - request REQ_DATA.body
        - requestParameters.PARAM_NAME VAL
        - requestMessage REQ_DATA.description
  - pathParams OPTS.pathParams
  - query|path|header|form|bodyParameters PARAM_ARR (using PATHDEF|OPERATION.parameters)
  - schema RESP.schema
  - noSchema BOOL (whether RESP.schema undefined)
  - path STR (full path with 'superagent', full URL with 'request') (using SPEC.basePath|schemes|host) (def: http://localhost:10010)
  - contentType OPERATION|SPEC.consumes
  - returnType OPERATION|SPEC.produces

  - headerApiKey|queryApiKey { type: 'VAR', name: 'SEC_NAME' }

  - isLoadTest true

                                 ##  - for each test:
                                 ##     - dot green|red according to "[not] ok"
                                 ##  - final summary:
                                 ##     - number of tests total|passed|failed
                                 ##     - errors: print each test 'TITLE' and YAML error props (as is)
                                 ##Parses with tap-out
                                 ##Sets exit code 1 if any failed test
dot                              ##Uses stdin|stdout


TAP-SPEC ==>                     ##4.1.1
                                 ##Mocha-spec-like:
                                 ##  - for each test:
                                 ##     - 'TITLE'
                                 ##     - green|red|cyan if ok|no ok|skip
                                 ##     - YAML error props (as is)
                                 ##  - for each FULL_TEST, comment or unknown line: printed as is
                                 ##  - final summary:
                                 ##     - number of tests total|passed|failed
                                 ##     - each failed test 'TITLE', grouped by FULL_TEST
                                 ##     - duration
                                 ##Parses with tap-out
                                 ##Sets exit code 1 if any failed test
tap-spec                         ##Uses stdin|stdout
TAP-SPEC([OPTS])->IOSTREAM       ##OPTS:
                                 ##  - padding STR (def: '  ')

TAP-DIFFLET ==>                  ##0.7.1
                                 ##Mocha-spec-like with diff:
                                 ##  - for each test:
                                 ##     - 'TITLE'
                                 ##     - green|red if ok|not ok
                                 ##     - YAML error props:
                                 ##        - expected|wanted and actual|found: line-wise colored diff
                                 ##        - "at": pretty printed
                                 ##        - others: as is
                                 ##     - only failed tests if OPTS.pessimistic true (def: false)
                                 ##  - for each FULL_TEST, comment (except # tests|pass|fail|ok) or unknown line: as is
                                 ##  - final summary:
                                 ##     - number of tests passed|failed
                                 ##     - duration
                                 ##Parses with tap-parser-yaml
                                 ##Sets exit code 1 if any failed test
tap-difflet                      ##Uses stdin|stdout
--pessimistic|p                  ##
TAP-DIFFLET([OPTS])->IOSTREAM    ##
